99       public static boolean isValidTimeRange(String str) {
100          if (!str.matches("\\d{2}:\\d{2}-\\d{2}:\\d{2}")){//"\\d{2} means exactly 2 digits
101              return false;
102          }
103          String[] parts = str.split("-");
104          String openTime = parts[0];
105          String closeTime = parts[1];
106  
107          try{
108              int openHour = Integer.parseInt(openTime.substring(0,2));
109              int openMin = Integer.parseInt(openTime.substring(3));
110              int closeHour = Integer.parseInt(closeTime.substring(0,2));
111              int closeMin = Integer.parseInt(closeTime.substring(3));
112  
113              return  openHour   >= 0 && openHour < 24 &&
114                      openMin    >= 0 && openMin  < 60 &&
115                      closeHour  >= 0 && closeHour < 24 &&
116                      closeMin   >= 0 && closeMin  < 60;
117          } catch (Exception e){
118              return false;
119          }
120      }

125      public static boolean isValidDate(String dateStr){
126          if (dateStr == null || dateStr.isEmpty()) return false;
127  
128          SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy", Locale.getDefault());
129          sdf.setLenient(false); // strict parsing
130          try {
131              sdf.parse(dateStr);
132              return true;
133          } catch (ParseException e) {
134              return false;
135          }
136      }
