81       public static int isValidHrMin(String timeStr) {    // --- Basic Checks ---
82           if (timeStr == null || timeStr.isEmpty()) {
83               return EMPTY_TIME;
84           }
85   
86           String lowerStr = timeStr.toLowerCase();
87   
88           // --- Check for 'h' ---
89           if (!lowerStr.contains("h")) {
90               return MISSING_H;
91           }
92   
93           // Validate minutes=
94           if (!lowerStr.contains("m")) {
95               return MISSING_M;
96           }
97   
98           // --- Split Hours & Minutes ---
99           String[] parts = lowerStr.split("h");
100          if (parts.length != 2) {
101              return INVALID_FORMAT;
102          }
103  
104          // --- Parse Hours ---
105          int hours;
106          try {
107              hours = Integer.parseInt(parts[0]);
108          } catch (NumberFormatException e) {
109              return HOURS_NOT_NUMBER;
110          }
111  
112          // --- Parse Minutes ---
113          String minutePart = parts[1].replace("m", ""); // Remove 'm' if present
114          int minutes;
115          try {
116              minutes = minutePart.isEmpty() ? 0 : Integer.parseInt(minutePart);
117          } catch (NumberFormatException e) {
118              return MINUTES_NOT_NUMBER;
119          }
120  
121          // --- Validate Ranges ---
122          if (hours < 0) {
123              return HOURS_NEGATIVE;
124          }
125          if (minutes < 0 || minutes > 59) {
126              return MINUTES_INVALID;
127          }
128  
129          return VALID_TIME; // All checks passed!
130      }
