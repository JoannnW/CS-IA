134      private Owner collectOwnerInpt(){
135          String storeName = storeNameInpt.getText().toString().trim();
136          String openingHours = openingHoursInpt.getText().toString().trim();
137  
138          double weight = -1;  // default invalid value
139          try {
140              String weightStr = weightInpt.getText().toString().trim().toLowerCase().replace(" ","");
141              weight = Double.parseDouble(weightStr.replace("kg", "").replace(",", "."));
142          } catch (NumberFormatException e) {
143              // weight remains -1
144          }
145  
146          double dailyIntake = -1;
147          try {
148              String dailyIntakeStr = dailyIntakeInpt.getText().toString().trim().toLowerCase().replace(" ","");
149              dailyIntake = Double.parseDouble(dailyIntakeStr.replace("kg", "").replace(",", "."));
150          } catch (NumberFormatException e) {
151              // dailyIntake remains -1
152          }
153  
154  
155          String latestShoppingDate = latestShoppingDateInpt.getText().toString().trim();
156  
157          ArrayList<String> daysOpen = new ArrayList<>();
158          //collect checkboxes
159          if (((CheckBox)findViewById(R.id.checkBox)).isChecked()) daysOpen.add("MON");
160          if (((CheckBox)findViewById(R.id.checkBox2)).isChecked()) daysOpen.add("TUE");
161          if (((CheckBox)findViewById(R.id.checkBox3)).isChecked()) daysOpen.add("WED");
162          if (((CheckBox)findViewById(R.id.checkBox4)).isChecked()) daysOpen.add("THU");
163          if (((CheckBox)findViewById(R.id.checkBox5)).isChecked()) daysOpen.add("FRI");
164          if (((CheckBox)findViewById(R.id.checkBox6)).isChecked()) daysOpen.add("SAT");
165          if (((CheckBox)findViewById(R.id.checkBox7)).isChecked()) daysOpen.add("SUN");
166          if (((CheckBox)findViewById(R.id.checkBox8)).isChecked()) daysOpen.add("Public Holidays");
167  
168          return new Owner(username, storeName, openingHours, weight, dailyIntake, latestShoppingDate, daysOpen, isGoogleConnected);
169      }

171      private void onSubmit(Owner owner){
172          //check Google sign in
173          if (!isGoogleConnected){
174              Toast.makeText(this, "Please connect your Google account", Toast.LENGTH_SHORT).show();
175              return;
176          }
177  
178          //create FirebaseOwner with all fields
179          FirebaseOwner firebaseOwner = new FirebaseOwner(
180                  owner.getName(),
181                  owner.getStoreName(),           //Store name
182                  owner.getOpeningHours(),           //Opening Hrs
183                  owner.getWeight(),                 //Total food wiehgt
184                  owner.getDailyIntake(),            //Daily intake
185                  owner.getLatestShoppingDate(),     //latest shopping date
186                  owner.getOpenDays());              //Open days
187  
188          //Add Google Sign-in status to Firebase data
189          firebaseOwner.setGoogleConnected(true);
190  
191          //save to Firebase - only if its validated and get reference
192          DatabaseReference usersRef = FirebaseDatabase.getInstance().getReference("users");
193          DatabaseReference newOwnerRef = usersRef.push();
194          newOwnerRef.setValue(firebaseOwner);
195          String ownerKey = newOwnerRef.getKey();
196  
197          //proceed to home page with key for future updates
198          Intent intent = new Intent(this, OwnerHome.class);
199          intent.putExtra("username", owner.getName());
200          intent.putExtra("storeName", owner.getStoreName());
201          intent.putExtra("openingHours", owner.getOpeningHours());
202          intent.putExtra("weight", owner.getWeight());
203          intent.putExtra("dailyIntake", owner.getDailyIntake());
204          intent.putExtra("latestShoppingDate", owner.getLatestShoppingDate());
205          intent.putStringArrayListExtra("openDays", new ArrayList<>(owner.getOpenDays()));
206          intent.putExtra("ownerKey", ownerKey); //needed for rescheduling
207  
208          startActivity(intent);
209          finish();
210      }

212      private boolean validateOwner(Owner owner){
213          areFieldsValid = false;
214          //field validations
215          //check if any slots are empty/ no checkboxes are checked
216          if(owner.getStoreName().isEmpty()){
217              Toast.makeText(this, "Store name is required", Toast.LENGTH_SHORT).show();
218              return false;
219          }
220          if (!Owner.isValidTimeRange(owner.getOpeningHours())){
221              Toast.makeText(this, "Please enter a valid time range (e.g. 09:00-17:00)", Toast.LENGTH_SHORT).show();
222              return false;
223          }
224          if (owner.getWeight() == -1 || !Owner.isValidWeight(owner.getWeight())){
225              Toast.makeText(this, "Weight must be between 0.00-999.99kg",Toast.LENGTH_SHORT).show();
226              return false;
227          }
228          if (owner.getDailyIntake() == -1 || !Owner.isValidWeight(owner.getDailyIntake())){
229              Toast.makeText(this, "Weight must be between 0.00-999.99kg",Toast.LENGTH_SHORT).show();
230              return false;
231          }
232          if (owner.getOpenDays().isEmpty()){
233              Toast.makeText(this, "Please select at least one open day", Toast.LENGTH_SHORT).show();
234              return false;
235          }
236          if (!Owner.isValidDate(owner.getLatestShoppingDate())){
237              Toast.makeText(this, "Please enter a valid date (e.g. 30/04/2008)", Toast.LENGTH_SHORT).show();
238              return false;
239          }
240  
241          areFieldsValid = true;
242          return areFieldsValid;
243      }
