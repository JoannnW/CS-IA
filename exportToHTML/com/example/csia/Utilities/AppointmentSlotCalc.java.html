17       public static ArrayList<String> findAvailableSlot(ArrayList<BusyTime> ownerBusyTimes,
18                                                    Object serviceProvider,
19                                                    ArrayList<String> openDays,
20                                                    int durationMin){
21           ArrayList<String> availableSlots = new ArrayList<>();
22           Calendar calendar = Calendar.getInstance();
23   
24           //check if the provider is available/ taken (linked)
25           if (serviceProvider instanceof FirebaseDoctor) {
26               Boolean isAvailable = ((FirebaseDoctor) serviceProvider).getLinked();
27               if (isAvailable != null && !isAvailable) {
28                   return availableSlots; // Return empty list if not available
29               }
30           } else if (serviceProvider instanceof FirebaseGroomer) {
31               Boolean isAvailable = ((FirebaseGroomer) serviceProvider).getLinked();
32               if (isAvailable != null && !isAvailable) {
33                   return availableSlots; // Return empty list if not available
34               }
35           }
36   
37           //get business hours
38           String businessHours = "09:00-17:00";
39           if (serviceProvider instanceof FirebaseDoctor){
40               String hrs = ((FirebaseDoctor) serviceProvider).businessHrs;
41               if (hrs != null && !hrs.trim().isEmpty()){
42                   businessHours = hrs;
43               }
44           } else if (serviceProvider instanceof FirebaseGroomer){
45               String hrs = ((FirebaseGroomer) serviceProvider).businessHrs;
46           if (hrs != null && !hrs.trim().isEmpty()){
47               businessHours = hrs;
48           }
49           }
50   
51           //split the business hours string into start and end times
52           String[] hours = businessHours.split("-");
53           if (hours.length < 2) return availableSlots;
54   
55           SimpleDateFormat timeFormat = new SimpleDateFormat("HH:mm", Locale.getDefault());
56           SimpleDateFormat slotFormat = new SimpleDateFormat("EEE, MMM d, yyyy HH:mm", Locale.getDefault());
57   
58           try {
59               //parse the start and end times into Date objects
60               Date openTime = timeFormat.parse(hours[0].trim());
61               Date closeTime = timeFormat.parse(hours[1].trim());
62   
63               //extract hour/min from parsed times into separate Cal instances
64               Calendar openCal = Calendar.getInstance();
65               openCal.setTime(openTime);
66   
67               Calendar closeCal = Calendar.getInstance();
68               closeCal.setTime(closeTime);
69   
70               //Loop through the next 7 days (starting from tmr)
71               for (int i = 0; i < 7; i ++){
72                   calendar.add(Calendar.DAY_OF_MONTH, 1);
73                   String dayOfWeek = calendar.getDisplayName(Calendar.DAY_OF_WEEK, Calendar.SHORT, Locale.US).toUpperCase();
74   
75                   //check if provider is open this day
76                   if (!openDays.contains(dayOfWeek)) continue;
77   
78                   //set calendar to open times
79                   Calendar slotCalendar = (Calendar) calendar.clone();
80                   slotCalendar.set(Calendar.HOUR_OF_DAY, openCal.get(Calendar.HOUR_OF_DAY));
81                   slotCalendar.set(Calendar.MINUTE, openCal.get(Calendar.MINUTE));
82                   slotCalendar.set(Calendar.SECOND, 0);
83                   slotCalendar.set(Calendar.MILLISECOND, 0);
84   
85                   //set the closing time for that same day
86                   Calendar dayClose = (Calendar) calendar.clone();
87                   dayClose.set(Calendar.HOUR_OF_DAY, closeCal.get(Calendar.HOUR_OF_DAY));
88                   dayClose.set(Calendar.MINUTE, closeCal.get(Calendar.MINUTE));
89                   dayClose.set(Calendar.SECOND, 0);
90                   dayClose.set(Calendar.MILLISECOND, 0);
91   
92                   //Generate open slots for this day
93                   while (slotCalendar.before(dayClose)){
94                       Calendar endCalendar = (Calendar) slotCalendar.clone();
95                       endCalendar.add(Calendar.MINUTE, durationMin);
96   
97                       //check if slot is available in owner's calendar
98                      if (isSlotAvailable(slotCalendar.getTime(), endCalendar.getTime(), ownerBusyTimes)){
99                          availableSlots.add(slotFormat.format(slotCalendar.getTime()));
100                     }
101  
102                      //Move to next slot by the appointment duration to find non-overlapping slots
103                      slotCalendar.add(Calendar.MINUTE, durationMin);
104  
105                  }
106  
107              }
108          } catch (ParseException e){
109              e.printStackTrace();
110          }
111  
112          ArrayList<String> topSlots = new ArrayList<>();
113          int count = Math.min(3, availableSlots.size());//in case there are less than 3 available slots
114          for (int i = 0; i < count; i++) {
115              topSlots.add(availableSlots.get(i));
116          }
117          //Return top 3 slots
118          return topSlots;
119      }

121      private static boolean isSlotAvailable(Date start, Date end, ArrayList<BusyTime> busyTimes){
122          for (BusyTime busy : busyTimes){
123             if (start.before(busy.getEndTime()) && end.after(busy.getStartTime())){
124                 return false; //there is a conflict
125             }
126          }
127          return true; //no conflict, is available
128      }
