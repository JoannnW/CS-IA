184      private Doctor collectDoctorInpt(){
185  
186          String businessHrs = businessHrsInpt.getText().toString().trim();
187  
188          durationStr = durationInpt.getText().toString().trim().toLowerCase().replace(" ", "");
189          int durationMin = 0;
190  
191          try {
192              int hIndex = durationStr.indexOf("h");
193              int mIndex = durationStr.indexOf("m");
194  
195              if (hIndex != -1) {
196                  String hoursStr = durationStr.substring(0, hIndex);
197                  durationMin += Integer.parseInt(hoursStr) * 60;
198              }
199  
200              if (mIndex != -1 && mIndex > hIndex) {
201                  String minutesStr = durationStr.substring(hIndex + 1, mIndex);
202                  durationMin += Integer.parseInt(minutesStr);
203              }
204          } catch (NumberFormatException e) {
205              Toast.makeText(this, "Invalid duration format", Toast.LENGTH_SHORT).show();
206          }
207  
208          ArrayList<String> daysOpen = new ArrayList<>();
209          //collect checkboxes
210          if (((CheckBox)findViewById(R.id.checkBox)).isChecked()) daysOpen.add("MON");
211          if (((CheckBox)findViewById(R.id.checkBox2)).isChecked()) daysOpen.add("TUE");
212          if (((CheckBox)findViewById(R.id.checkBox3)).isChecked()) daysOpen.add("WED");
213          if (((CheckBox)findViewById(R.id.checkBox4)).isChecked()) daysOpen.add("THU");
214          if (((CheckBox)findViewById(R.id.checkBox5)).isChecked()) daysOpen.add("FRI");
215          if (((CheckBox)findViewById(R.id.checkBox6)).isChecked()) daysOpen.add("SAT");
216          if (((CheckBox)findViewById(R.id.checkBox7)).isChecked()) daysOpen.add("SUN");
217          if (((CheckBox)findViewById(R.id.checkBox8)).isChecked()) daysOpen.add("Public Holidays");
218          return new Doctor(username, daysOpen, businessHrs, durationMin, isGoogleConnected);
219      }

254      private boolean validateDoctor(Doctor doctor){
255          areFieldsValid = false;
256          int status = Doctor.isValidHrMin(durationStr);
257  
258          //field validations
259          //check if any slots are empty/ no checkboxes are checked
260          if(doctor.getOpenDays().isEmpty()){
261              Toast.makeText(this, "Please fill out all fields", Toast.LENGTH_SHORT).show();
262              return false;
263          }
264          if (!Doctor.isValidTimeRange(doctor.getBusinessHrs())){
265              Toast.makeText(this, "Please enter a valid time range (e.g. 09:00-17:00)", Toast.LENGTH_SHORT).show();
266              return false;
267          }
268          if (status == Doctor.VALID_TIME) {
269              areFieldsValid = true;
270              return areFieldsValid;
271          } else {
272              String errorMessage;
273  
274              if (status == Doctor.EMPTY_TIME) {
275                  errorMessage = "Duration cannot be empty";
276              }
277              else if (status == Doctor.MISSING_H) {
278                  errorMessage = "Missing 'h' in duration (e.g., 2h30m)";
279              }
280              else if (status == Doctor.INVALID_FORMAT) {
281                  errorMessage = "Invalid format - use like 2h30m";
282              }
283              else if (status == Doctor.HOURS_NOT_NUMBER) {
284                  errorMessage = "Hours must be a number";
285              }
286              else if (status == Doctor.MINUTES_NOT_NUMBER) {
287                  errorMessage = "Minutes must be a number";
288              }
289              else if (status == Doctor.HOURS_NEGATIVE) {
290                  errorMessage = "Hours cannot be negative";
291              }
292              else if (status == Doctor.MINUTES_INVALID) {
293                  errorMessage = "Minutes must be between 00-59";
294              }
295              else if (status == Doctor.MISSING_M) {
296                  errorMessage = "Missing 'm' in duration";
297              }
298              else {
299                  errorMessage = "Invalid duration format";
300              }
301  
302              Toast.makeText(this, errorMessage, Toast.LENGTH_SHORT).show();
303              return false;
304          }
305      }
