102      private void displaySlots(ArrayList<String> slots) {

106  
107              if (i < slots.size()) {
108                  String slot = slots.get(i);
109                  slotText.setText("Option " + (i + 1) + ": " + slot);
110                  deferBtn.setEnabled(true);

112                  int finalI = i;
113                  deferBtn.setOnClickListener(v -> deferSlot(slot, finalI));

115                  slotText.setText("No slot");

121      private void deferSlot(String originalSlot, int index) {
122          Calendar calendar = Calendar.getInstance();
123          new DatePickerDialog(this, (view, year, month, day) -> {
124              calendar.set(year, month, day);
125              new TimePickerDialog(this, (timeView, hour, minute) -> {
126                  calendar.set(Calendar.HOUR_OF_DAY, hour);
127                  calendar.set(Calendar.MINUTE, minute);
128  
129                  SimpleDateFormat sdf = new SimpleDateFormat("dd/MM/yyyy HH:mm", Locale.getDefault());
130                  String newSlot = sdf.format(calendar.getTime());
131  
132                  // Create the query properly
133                  DatabaseReference ref = FirebaseDatabase.getInstance()
134                          .getReference("appointment_requests");
135                  Query query = ref.orderByChild("requestedOption/dateTimeRange")
136                          .equalTo(originalSlot);
137  
138                  query.addListenerForSingleValueEvent(new ValueEventListener() {
139                      @Override
140                      public void onDataChange(@NonNull DataSnapshot snapshot) {
141                          for (DataSnapshot ds : snapshot.getChildren()) {
142                              // Use ds.getRef() to get the reference to the specific node
143                              DatabaseReference requestRef = ds.getRef();
144                              requestRef.child("requestedOption/dateTimeRange").setValue(newSlot);
145                              requestRef.child("requestedOption/isDeferred").setValue(true);
146  
147                              // Update UI
148                              TextView slotText = findViewById(textIds[index]);
149                              slotText.setText("Option " + (index + 1) + ": " + newSlot);
150                              Toast.makeText(GroomerHome.this, "Slot deferred", Toast.LENGTH_SHORT).show();
151                          }
152                      }
153  
154                      @Override
155                      public void onCancelled(@NonNull DatabaseError error) {
156                          Toast.makeText(GroomerHome.this, "Failed to defer", Toast.LENGTH_SHORT).show();
157                      }
158                  });
159              }, 12, 0, false).show();
160          },
161                  calendar.get(Calendar.YEAR),
162                  calendar.get(Calendar.MONTH),
163                  calendar.get(Calendar.DAY_OF_MONTH))
164                  .show();
165      }

178      private void loadIncomingRequest() {
179          DatabaseReference ref = FirebaseDatabase.getInstance()
180                  .getReference("appointment_requests");
181          Query query = ref.orderByChild("providerName")
182                  .equalTo(groomer.getName());
183  
184          ref.addValueEventListener(new ValueEventListener() {
185              @Override
186              public void onDataChange(@NonNull DataSnapshot snapshot) {
187                  for (DataSnapshot ds : snapshot.getChildren()) {
188                      AppointmentRequest request = ds.getValue(AppointmentRequest.class);
189                      if (request != null && "pending".equals(request.getStatus())) { //Owner made a request
190                          // Show request
191                          TextView requestTxt = findViewById(R.id.textView31);
192                          requestTxt.setText("Requested: " + request.getRequestedOption().getDateTimeRange());
193  
194                          // Hide defer buttons
195                          int[] deferBtns = {R.id.button11, R.id.button4, R.id.button12};
196                          for (int btn : deferBtns) findViewById(btn).setVisibility(View.GONE);
197  
198                          // Show accept/reject buttons
199                          Button acceptBtn = findViewById(R.id.button6);
200                          Button rejectBtn = findViewById(R.id.button7);
201                          acceptBtn.setVisibility(View.VISIBLE);
202                          rejectBtn.setVisibility(View.VISIBLE);
203  
204                          // Set click listeners
205                          acceptBtn.setOnClickListener(v -> {
206                              ds.getRef().child("status").setValue("accepted");
207                              acceptBtn.setEnabled(false);
208                              rejectBtn.setEnabled(false);
209                          });
210  
211                          rejectBtn.setOnClickListener(v -> {
212                              ds.getRef().child("status").setValue("rejected");
213                              acceptBtn.setEnabled(false);
214                              rejectBtn.setEnabled(false);
215                          });
216                      }
217                  }
218              }
219  
220              @Override
221              public void onCancelled(@NonNull DatabaseError error) {
222                  Toast.makeText(GroomerHome.this, "Failed to load requests", Toast.LENGTH_SHORT).show();
223              }
224          });
225      }

227      private void checkAvailabilityStatus() {
228          if (groomerKey != null) {
229              DatabaseReference groomerRef = FirebaseDatabase.getInstance()
230                      .getReference("users")
231                      .child(groomerKey)
232                      .child("available");
233  
234              groomerRef.addListenerForSingleValueEvent(new ValueEventListener() {
235                  @Override
236                  public void onDataChange(@NonNull DataSnapshot snapshot) {
237                      if (snapshot.exists()) {
238                          boolean isAvailable = snapshot.getValue(Boolean.class);
239                          availabilitySwitch.setChecked(isAvailable);
240                      } else {
241                          // Default to available if not set
242                          availabilitySwitch.setChecked(true);
243                      }
244                  }
245  
246                  @Override
247                  public void onCancelled(@NonNull DatabaseError error) {
248                      Toast.makeText(GroomerHome.this, "Failed to check availability", Toast.LENGTH_SHORT).show();
249                  }
250              });
251          }
252      }

254      private void updateAvailabilityStatus(boolean isChecked, FirebaseGroomer groomer) {
255          if (groomerKey != null) {
256              DatabaseReference groomerRef = FirebaseDatabase.getInstance()
257                      .getReference("users")
258                      .child(groomerKey)
259                      .child("available");
260  
261              groomerRef.setValue(isChecked)
262                      .addOnSuccessListener(aVoid -> {
263                          String status = isChecked ? "available" : "unavailable";
264                          Toast.makeText(GroomerHome.this, "You're now " + status, Toast.LENGTH_SHORT).show();
265                          groomer.setLinked(isChecked);
266                      })
267                      .addOnFailureListener(e -> {
268                          Toast.makeText(GroomerHome.this, "Failed to update availability", Toast.LENGTH_SHORT).show();
269                      });
270          }
271      }
